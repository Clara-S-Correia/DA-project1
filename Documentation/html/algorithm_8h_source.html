56bool BFS_part(Graph<T> *g, Vertex<T> *s, Vertex<T> *t) {
57 // Mark all vertices as not visited
58 for(auto v : g->getVertexSet()) {
59 v->setVisited(false);
60 v->setPath(nullptr);
61 }
62
63 // Mark the source vertex as visited and enqueue it
64 s->setVisited(true);
65 std::queue<Vertex<T> *> q;
66 q.push(s);
67
68 // BFS to find an augmenting path
69 while(!q.empty()) {
70 auto v = q.front();
71 q.pop();
72
73 for(auto e: v->getAdj()) {
74 auto w = e->getDest();
75 //residual capacity: Capacity-Flow > 0
76 auto residual = e->getCapacity()- e->getFlow();
77 if (w != nullptr && !w->isVisited() && residual>0) {
78 w->setVisited(true);
79 w->setPath(e);
80 q.push(w);
81 //if path is made to the sink(T)
82 if (w == t) return true;
83 }
84 }
85 }
86
87 // Return true if a path to the target is found, false otherwise
88 return t->isVisited();
89}

104void Edmonds_karp(Graph<T> *g, T source, T sink) {
105 Vertex<T> *s = g->findVertex(source);
106 Vertex<T> *t = g->findVertex(sink);
107
108 //in case parser fails, findVertex will return nullptr thuis checking

111 //while bfs(graph, source, sink)
112 while (BFS_part(g, s, t)) {
113 double n_min = INF;
114
115 //looking for n_max capacity of path by bfs
116 for (auto v=t ; v!=s; ) {
117 //starting from sink, until the source
118 auto e = v->getPath();//get edge of vertex v
119 n_min = min(n_min, e->getCapacity()-e->getFlow());//getting smaller one, the current max or the path found
120 v=e->getOrig();//moving backwards - getting the original
121 }
122
123 for (auto v=t ; v!=s; ) {
124 auto e = v->getPath();
125 //adding flow to the forward edge
126 e->setFlow(e->getFlow() + n_min);
127 //setting flow to -> the initial flow at the edge+cureent min path
128
129 //minus flow from the prev edge
130 auto rev = e->getReverse();
131 if (rev!=nullptr) rev->setFlow(rev->getFlow() - n_min);
132 //setting flow to-> the prev flow-min path
133

135 }
